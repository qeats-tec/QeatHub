-- Gelişmiş Hız ve Hava İvmesi (Qeat Altyapı Motoru)
game:GetService("RunService").Heartbeat:Connect(function()
    if Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") then
        local hum = Player.Character:FindFirstChildOfClass("Humanoid")
        local rootPart = Player.Character.HumanoidRootPart
        if not hum or not rootPart then return end
        
        if hum.MoveDirection.Magnitude > 0 then
            local moveDirection = hum.MoveDirection
            
            -- Gelişmiş İvmeli Hız Fizik Desteği (Yerdeyken)
            if hum.FloorMaterial ~= Enum.Material.Air and CurrentWalkSpeed > 16 then
                rootPart.AssemblyLinearVelocity = Vector3.new(
                    moveDirection.X * CurrentWalkSpeed,
                    rootPart.AssemblyLinearVelocity.Y,
                    moveDirection.Z * CurrentWalkSpeed
                )
            end
            
            -- Gelişmiş Hava İvmesi (Air Strafe - Havadayken)
            if AirStrafeEnabled and (hum:GetState() == Enum.HumanoidStateType.Jumping or hum:GetState() == Enum.HumanoidStateType.Freefall) then
                local currentVelocity = rootPart.AssemblyLinearVelocity
                local targetVelocityX = moveDirection.X * (CurrentWalkSpeed * AirStrafeValue)
                local targetVelocityZ = moveDirection.Z * (CurrentWalkSpeed * AirStrafeValue)
                
                rootPart.AssemblyLinearVelocity = Vector3.new(
                    math.clamp(currentVelocity.X + (targetVelocityX * 0.1), -250, 250),
                    currentVelocity.Y,
                    math.clamp(currentVelocity.Z + (targetVelocityZ * 0.1), -250, 250)
                )
            end
        end
    end
end)

-- Gelişmiş Zıplama Yardımı (Bhop)

end)

local AimbotEnabled = false
local HitboxSize = 2

local function isVisible(targetPart)
    local char = Player.Character
    if not char or not char:FindFirstChild("Head") then return false end
    local origin = Camera.CFrame.Position
    local direction = (targetPart.Position - origin).Unit * (targetPart.Position - origin).Magnitude
    local ray = Ray.new(origin, direction)
    local hitPart, hitPosition = game.Workspace:FindPartOnRayWithIgnoreList(ray, {char, Camera})
    if hitPart and hitPart:IsDescendantOf(targetPart.Parent) then return true end
    return false
end

local function getClosestPlayer()
    local closest, dist = nil, math.huge
    for _, p in pairs(game.Players:GetPlayers()) do
        if p ~= Player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") and p.Character:FindFirstChild("Head") then
            local pos, onScreen = Camera:WorldToViewportPoint(p.Character.HumanoidRootPart.Position)
            if onScreen then
                if isVisible(p.Character.Head) then
                    local mouseDist = (Vector2.new(Mouse.X, Mouse.Y) - Vector2.new(pos.X, pos.Y)).Magnitude
                    if mouseDist < dist then
                        closest = p
                        dist = mouseDist
                    end
                end
            end
        end
    end
    return closest
end

game:GetService("RunService").RenderStepped:Connect(function()
    if AimbotEnabled and game:GetService("UserInputService"):IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
        local target = getClosestPlayer()
        if target and target.Character and target.Character:FindFirstChild("Head") then
            Camera.CFrame = CFrame.new(Camera.CFrame.Position, target.Character.Head.Position)
        end
    end
end)
